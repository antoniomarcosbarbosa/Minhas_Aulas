\chapter{Produtos Notáveis}

Produtos notáveis são multiplicações algébricas que ocorrem com frequência. 

\section{Quadrado da soma}

\begin{equation}
(a+b)^2=a^2+2ab+b^2
\end{equation}

\section{Quadrado da Diferença}

\begin{equation}
(a-b)^2=a^2-2ab+b^2
\end{equation}

\section{Produto da soma pela diferença}

\begin{equation}
(a+b)(a-b)=a^2+b^2
\end{equation}
